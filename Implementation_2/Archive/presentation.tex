\section{Optimization Models}
%========================================

\begin{frame}
    \frametitle{Gradient Descent Framework}
    \begin{block}{General Update Rule}
        $$\mathbf{w}_{t+1} = \mathbf{w}_t - \alpha \nabla f(\mathbf{w}_t)$$
    \end{block}
    \begin{itemize}
        \item $\mathbf{w}$: coefficient vector
        \item $\alpha$: learning rate
        \item $\nabla f(\mathbf{w})$: gradient of objective function
    \end{itemize}
    \vspace{0.5cm}
    \textbf{Hyperparameters:}
    \begin{itemize}
        \item Learning rate: $\alpha = 0.01$
        \item Convergence tolerance: $\text{tol} = 10^{-4}$
        \item Maximum iterations: $\text{max\_iter} = 1000$
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{Model 1: Baseline (MSE Only)}
    \begin{block}{Objective Function}
        $$f(\mathbf{w}) = \frac{1}{2n}\sum_{i=1}^{n} (\mathbf{x}_i^T\mathbf{w} - y_i)^2$$
    \end{block}
    \begin{block}{Gradient}
        $$\nabla f(\mathbf{w}) = \frac{1}{n}\mathbf{X}^T(\mathbf{X}\mathbf{w} - \mathbf{y})$$
    \end{block}
    \begin{block}{Update Rule}
        $$\mathbf{w}_{t+1} = \mathbf{w}_t - \alpha \cdot \frac{1}{n}\mathbf{X}^T(\mathbf{X}\mathbf{w}_t - \mathbf{y})$$
    \end{block}
    \textbf{Purpose:} Baseline model with no regularization constraints
\end{frame}

\begin{frame}[fragile]
    \frametitle{Model 2: L1 Regularization (Lasso)}
    \begin{block}{Objective Function}
        $$f(\mathbf{w}) = \frac{1}{2n}\sum_{i=1}^{n} (\mathbf{x}_i^T\mathbf{w} - y_i)^2 + \lambda \|\mathbf{w}_{1:}\|_1$$
    \end{block}
    \begin{block}{Soft-Thresholding Operator}
        $$S_{\lambda}(z_j) = \begin{cases} 
            z_j - \lambda & \text{if } z_j > \lambda \\
            0 & \text{if } |z_j| \leq \lambda \\
            z_j + \lambda & \text{if } z_j < -\lambda
        \end{cases}$$
    \end{block}
    \begin{block}{Update Rule}
        $$\mathbf{z} = \mathbf{w}_t - \alpha \nabla f(\mathbf{w}_t)$$
        $$\mathbf{w}_{t+1} = \text{sign}(\mathbf{z}) \odot \max(|\mathbf{z}| - \alpha\lambda, 0)$$
    \end{block}
    \textbf{Hyperparameter:} $\lambda = 0.1$ (regularization strength) \\
    \textbf{Purpose:} Feature selection through coefficient sparsity
\end{frame}

\begin{frame}[fragile]
    \frametitle{Model 3: L2 Regularization (Ridge)}
    \begin{block}{Objective Function}
        $$f(\mathbf{w}) = \frac{1}{2n}\sum_{i=1}^{n} (\mathbf{x}_i^T\mathbf{w} - y_i)^2 + \frac{\lambda}{2}\|\mathbf{w}_{1:}\|_2^2$$
    \end{block}
    \begin{block}{Gradient}
        $$\nabla f(\mathbf{w}) = \frac{1}{n}\mathbf{X}^T(\mathbf{X}\mathbf{w} - \mathbf{y}) + \lambda \mathbf{w}$$
    \end{block}
    \begin{block}{Update Rule}
        $$\mathbf{w}_{t+1} = \mathbf{w}_t - \alpha \left[\frac{1}{n}\mathbf{X}^T(\mathbf{X}\mathbf{w}_t - \mathbf{y}) + \lambda \mathbf{w}_t\right]$$
    \end{block}
    \textbf{Hyperparameter:} $\lambda = 0.1$ (regularization strength) \\
    \textbf{Purpose:} Coefficient shrinkage, prevents overfitting through weight decay
\end{frame}

\begin{frame}
    \frametitle{Regularization Comparison}
    \begin{table}
        \centering
        \small
        \begin{tabular}{lll}
            \toprule
            \textbf{Aspect} & \textbf{L1 (Lasso)} & \textbf{L2 (Ridge)} \\
            \midrule
            \textbf{Penalty Term} & $\lambda\|\mathbf{w}\|_1$ & $\lambda\|\mathbf{w}\|_2^2$ \\
            \textbf{Feature Selection} & Yes (sparse) & No (shrinkage) \\
            \textbf{Coefficient Behavior} & Exact zeros & Continuous shrinkage \\
            \textbf{Numerical Stability} & Moderate & High \\
            \textbf{Interpretability} & Feature elimination & Reduced importance \\
            \bottomrule
        \end{tabular}
    \end{table}
\end{frame}

\section{Convergence Analysis}
%========================================

\begin{frame}
    \frametitle{Convergence Criterion}
    \begin{block}{Weight Change Criterion}
        $$\|\mathbf{w}_{t+1} - \mathbf{w}_t\| < \text{tolerance}$$
    \end{block}
    \begin{itemize}
        \item Convergence tolerance: $10^{-4}$
        \item Monitors coefficient stability between iterations
        \item More robust than gradient norm criterion for L1 regularization
    \end{itemize}
    \vspace{0.5cm}
    \textbf{Implementation Details:}
    \begin{itemize}
        \item Bias term (first coefficient) excluded from regularization
        \item Regularization applied only to feature coefficients
        \item Allows bias to adapt independently
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Model Performance Metrics}
    \begin{block}{Mean Squared Error (MSE)}
        $$\text{MSE} = \frac{1}{m}\sum_{i=1}^{m} (y_{\text{pred},i} - y_{\text{actual},i})^2$$
    \end{block}
    \begin{itemize}
        \item Training error: Performance on training dataset
        \item Testing error: Generalization to unseen data
        \item Difference indicates potential overfitting
    \end{itemize}

    \textbf{Expected Results:}
    \begin{itemize}
        \item Baseline: Lower training, potentially higher testing error
        \item L1: Sparse coefficients, moderate generalization
        \item L2: Continuous shrinkage, best generalization expected

def gradient_descent(X, w, y, obFuncType='mse', lam=0, 
                     alpha=0.01, tol=1e-4, max_iter=1000):
    i = 1
    mse_vals = []
    iter_vals = []
    
    while (i <= max_iter):
        gradient, soft_thresh = gradFunc(X, w, y, obFuncType, lam)
        
        if obFuncType == 'l1':
            z = w - alpha * gradient
            w_new = np.sign(z) * np.maximum(np.abs(z) - alpha * lam, 0)
            w_new[0] = z[0]  # bias not regularized
        else:
            step_size = alpha * gradient
            w_new = w - step_size
        
        if np.linalg.norm(w_new - w) < tol:
            break
        
        w = w_new
        mse_vals.append(objFunc(X, w, y, obFuncType, lam))
        i += 1
    
    return w, mse_vals, iter_vals

\section{Key Findings \& Conclusions}
%========================================

\begin{frame}
    \frametitle{Key Findings}
    \begin{enumerate}
        \item \textbf{Regularization Impact:} Both L1 and L2 reduce overfitting risk
        \item \textbf{Feature Selection:} L1 regularization identifies important features through sparsity
        \item \textbf{Model Stability:} L2 regularization provides more stable coefficients
        \item \textbf{Convergence Behavior:} All models converge within specified tolerance
        \item \textbf{Generalization:} Regularized models show better test performance
    \end{enumerate}
\end{frame}

\begin{frame}
    \frametitle{Conclusions}
    \begin{block}{Primary Insights}
        \begin{itemize}
            \item Custom gradient descent implementation successfully matches scikit-learn results
            \item Regularization techniques effectively balance fit and complexity
            \item L1 and L2 serve different purposes in model optimization
        \end{itemize}
    \end{block}
    \vspace{0.5cm}
    \begin{block}{Recommendations}
        \begin{itemize}
            \item Use L2 for stable predictions with all features
            \item Use L1 for interpretability and sparse feature sets
            \item Tune $\lambda$ parameter based on cross-validation
        \end{itemize}
    \end{block}
\end{frame}
